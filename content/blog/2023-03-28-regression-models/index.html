Segmented model

# dataset with categorical variable
dfseg <- 
  df |> 
  mutate(x = if_else(condition = Days <= 90,true = 0,false = 1),
         dif = Days - 90,
         X = dif * x)

# Segmetned Model
m3 <- lm(Prices ~ Days + X, dfseg)
summary(m3)
